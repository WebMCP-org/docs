```javascript
async execute({ param }) {
  try {
    const result = await performOperation(param);
    return {
      content: [{ type: "text", text: `Success: ${result}` }]
    };
  } catch (error) {
    console.error('Tool execution error:', error);
    return {
      content: [{
        type: "text",
        text: `Failed to execute: ${error.message}`
      }],
      isError: true
    };
  }
}
```
